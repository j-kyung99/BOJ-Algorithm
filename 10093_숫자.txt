#pragma warning(disable:4996)
#include <iostream>
#include <stdio.h>
#include <stdlib.h>
#include <algorithm>
#include <vector>

using namespace std;

int main(void) {
	long long a, b;
	long long count;
	cin >> a >> b;
	if (a > b) swap(a, b);
	if (a == b) cout << 0;
	else {
		count = b - a - 1;
		cout << count << '\n';
		for (long long i = a + 1; i < b; i++) {
			cout << i << " ";
		}
	}
}
/* 꽤 까다로운 친구네,, a와 b가 같을 때의 예외 설정 필수!
   그리고 int 크기가 한정되어 있어서 long long 쓰기! */
